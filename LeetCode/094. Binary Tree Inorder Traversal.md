```javascript
function inorderTraversal(root) {
  if (!root) return [];
  let nodes = [];
  let numbers = [];
  let curr = root;
  while (curr || nodes.length) {
    while (curr) {
      nodes.push(curr);
      curr = curr.left;
    }
    curr = nodes.pop();
    numbers.push(curr.val);
    curr = curr.right;
  }
  return numbers;
}
```
